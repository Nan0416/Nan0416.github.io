    1. Send HTTP GET

                          // url, [request data], [success-callback (data, string-status)], [expected response format]
                    // it returns a JQuery XHR object.
                    jqxhr = $.get(`realtime/trip/10`, null, null, "json");
                    
                    // the jqxhr object has three methods: .done, .fail, .always. Each one takes a callback function.
                    // .done invoked if the request is successful
                    // .fail ...
                    // .always is invoked in both cases.
                    jqxhr.done((data, status, jqxhr_)=>{
                        console.log(typeof(data)); // object
                        console.log(typeof(status)); // string = "success"
                        console.log(jqxhr == jqxhr_); // true
                    }).fail((jqxhr_, status, data)=>{
                        console.log(typeof(data)); // object
                        console.log(typeof(status)); // string = "success"
                        console.log(jqxhr == jqxhr_); // true
                    }).always((data, arg1, status, arg2)=>{
                        // arg1 and arg2 depends on status
                    })
