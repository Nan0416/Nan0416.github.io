  1. Introduction

    String's problems usually use other techniques to solve. e.g. Convert to char[] and then using two pointers.

    String problems:
    1). Longest Palindrome Substring. (Dynamic Programming) 2). Longest Non-repeated Substring. (Two Pointers) 3). Pure Manipulation (reverse a string, zigzag)

    When dealing with sub-array or sub-string, first defining a index model (left inclusive, right inclusive or left inclusive, right exculsive)

  2. Example DP

    1. Longest palindromic substring

      class Solution {
    public String longestPalindrome(String s) {
        if(s == null || s.length() == 0) return s;
        int [][] result = new int[s.length()][s.length()];
        int maxLen = 1;
        int startIndex = 0, endIndex = 0;
        for(int len = 1; len <= s.length(); len++){
            for(int i = 0; i + len <= s.length(); i++){
                if(len == 1){
                    result[i][i + len - 1] = 1;
                }else if(len == 2){
                    if(s.charAt(i) == s.charAt(i + len - 1)){
                        result[i][i + len - 1] = 1;
                        maxLen = 2;
                        startIndex = i;
                        endIndex = i + len - 1;
                    }else{
                        result[i][i + len - 1] = 0;
                    }
                }else{
                    if(s.charAt(i) == s.charAt(i + len - 1) 
                       && result[i + 1][i + len - 2] == 1){
                        result[i][i + len - 1] = 1;
                        if(len > maxLen){
                            maxLen = len;
                            startIndex = i;
                            endIndex = i + len - 1;
                        }
                    }else{
                        result[i][i + len - 1] = 0;
                    }
                }
            }
        }
        return s.substring(startIndex, endIndex + 1);
    }
}

  3. Example Pure

    1. ZigZag Conversion

      // corner case: numRows = 1
class Solution {
    public String convert(String s, int numRows) {
        if(numRows == 1) return s;
        Character [][] result = new Character[numRows][s.length()];
        char [] result2 = new char[s.length()];
        boolean risingEdge = false;
        int counter = 0;
        int index = 0;
        while(true){
            for(int i = 0; i < numRows; i++){
                if(counter == s.length()){
                    int tempC = 0;
                    for(int k = 0; k < numRows; k++){
                        for(int j = 0; j < s.length(); j++){
                            if(result[k][j] != null){
                                result2[tempC++] = result[k][j];
                            }
                        }
                    }
                    return new String(result2);
                }
                if(risingEdge){
                    // numRows >= 3
                    index++;
                    int temp = numRows - i - 2;
                    if(temp == 0){
                        break;
                    }
                    result[temp][index] = s.charAt(counter++);
                }else{
                    result[i][index] = s.charAt(counter++);
                }
            }
            risingEdge = !risingEdge;
        }
    }
}
